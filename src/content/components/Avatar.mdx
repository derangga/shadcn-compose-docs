## Basic Usage

<TabPreview>

<TabPreview.Panel name="Preview">
<Preview name="avatar" variant="default" />
</TabPreview.Panel>

<TabPreview.Panel name="Code">

```kotlin
import com.shadcn.ui.components.Avatar

@Composable
fun Example() {
    Column(
        modifier = Modifier
            .fillMaxSize()
            .padding(horizontal = 16.dp, vertical = 32.dp),
        horizontalAlignment = Alignment.CenterHorizontally
    ) {
        Avatar(
            imageUrl = "https://avatars.githubusercontent.com/u/124599?v=4",
            contentDescription = "Avatar",
            fallbackText = "CN"
        )
    }
}
```

import com.shadcn.ui.components.Avatar

@Composable

    ) {
        Avatar(
            imageUrl = "https://invalid-image-url.com/nonexistent.png",

import com.shadcn.ui.components.Avatar

@Composable
fun Example() {
    Column(
        modifier = Modifier
            .fillMaxSize()

        horizontalAlignment = Alignment.CenterHorizontally
    ) {
        Avatar(
            imageUrl = "https://picsum.photos/200/300",
            contentDescription = "Random Image",
            fallbackText = "RI",
            loadingContent = {
                CircularProgressIndicator(modifier = Modifier.size(24.dp), color = MaterialTheme.shadcnColors.primary)
            }
        )
    }
}
